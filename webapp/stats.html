async function loadStats() { try { const vilkiResponse = await fetch('https://arbitrageeerbot-backend-danik20021.koyeb.app/get-vilki'); const vilki = await vilkiResponse.json(); const vilkiTable = document.getElementById('vilki-table').getElementsByTagName('tbody')[0]; vilkiTable.innerHTML = ''; vilki.forEach(vilka => { const row = vilkiTable.insertRow(); row.insertCell().textContent = vilka.date; row.insertCell().textContent = vilka.event; row.insertCell().textContent = vilka.bk1; row.insertCell().textContent = vilka.outcome1; row.insertCell().textContent = vilka.odds1; row.insertCell().textContent = vilka.stake1; row.insertCell().textContent = vilka.bk2; row.insertCell().textContent = vilka.outcome2; row.insertCell().textContent = vilka.odds2_history; row.insertCell().textContent = vilka.stake2; row.insertCell().textContent = vilka.profit; }); const bonusesResponse = await fetch('https://arbitrageeerbot-backend-danik20021.koyeb.app/get-bonuses'); const bonuses = await bonusesResponse.json(); const bonusesTable = document.getElementById('bonuses-table').getElementsByTagName('tbody')[0]; bonusesTable.innerHTML = ''; bonuses.forEach(bonus => { const row = bonusesTable.insertRow(); row.insertCell().textContent = bonus.date; row.insertCell().textContent = bonus.bk; row.insertCell().textContent = bonus.type; row.insertCell().textContent = bonus.amount; row.insertCell().textContent = bonus.conditions; row.insertCell().textContent = bonus.deadline; row.insertCell().textContent = bonus.status; row.insertCell().textContent = bonus.profit; }); } catch (error) { console.error('Ошибка загрузки статистики:', error); } } window.onload = loadStats;
